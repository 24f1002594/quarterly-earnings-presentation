Customer Analytics Dashboard

Real-time metrics visualization


class AnalyticsDashboard {
    constructor(apiEndpoint) {
        this.endpoint = apiEndpoint;
        this.metrics = new Map();
    }
    
    async fetchMetrics() {
        const response = await fetch(`${this.endpoint}/metrics`);
        const data = await response.json();
        
        return data.map(metric => ({
            name: metric.name,
            value: metric.value,
            change: this.calculateChange(metric)
        }));
    }
    
    calculateChange(metric) {
        const previous = this.metrics.get(metric.name);
        if (!previous) return 0;
        
        return ((metric.value - previous) / previous) * 100;
    }
}

// Usage
const dashboard = new AnalyticsDashboard('/api/v2');
const metrics = await dashboard.fetchMetrics();
